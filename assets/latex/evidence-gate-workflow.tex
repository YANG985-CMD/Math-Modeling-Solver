\tikzset{
  every node/.style={font=\small, text=Ink},
  process/.style={

    fill=ProcessFill,
    line width=0.8pt,

    minimum height=11mm,
    text width=19mm,
    align=center,
    inner sep=2.2pt
  },
  gate/.style={
    diamond,
    aspect=1.65,
    draw=GateEdge,
    fill=GateFill,
    line width=0.9pt,
    minimum width=19mm,
    minimum height=11mm,
    align=center,
    inner sep=1.0pt,
    font=\scriptsize\bfseries
  },
  output/.style={
    process,
    draw=OutputEdge,
    fill=OutputFill,
    font=\small\bfseries
  },
  flow/.style={
    -{Stealth[length=2.2mm,width=1.6mm]},
    draw=Line,
    line width=0.9pt,
    rounded corners=2pt
  },
  failflow/.style={
    -{Stealth[length=2.0mm,width=1.4mm]},

    line width=0.75pt,

  },
  faildrop/.style={
    draw=FailEdge,
    line width=0.65pt,
    dashed
  },
  failbox/.style={
    rectangle,
    rounded corners=3pt,
    draw=FailEdge,
    fill=FailFill,
    line width=0.8pt,
    minimum width=63mm,
    minimum height=9.5mm,
    align=center,
    font=\small\bfseries
  },
  stage/.style={

    fill=BandFill,

  },
  stageLabel/.style={
    font=\scriptsize\bfseries,
    text=GateEdge,

    inner xsep=4pt,
    inner ysep=1.5pt,
    rounded corners=2pt
  }
}
